42template <double MinWeight = -std::numeric_limits<double>::infinity(),
43 double MaxWeight = std::numeric_limits<double>::infinity()>
44struct EdgeWeightValidator {
45 template <HasWeightOnEdges GraphType>
46 static void validate(const GraphType &graph) {
47 const auto [edges_begin, edges_end] = boost::edges(graph);
48
49 for (const auto &edge : boost::make_iterator_range(edges_begin, edges_end)) {
50 const double weight = graph[edge].weight;
51 auto source = boost::source(edge, graph);
52 auto target = boost::target(edge, graph);
53
54 if (weight < MinWeight || weight > MaxWeight) {
55 throw GraphValidationError(
56 "Invalid weight " + std::to_string(weight) +
57 " on edge from '" + graph[source].name + "' to '" +
58 graph[target].name + "' (must be in range [" +
59 std::to_string(MinWeight) + ", " + std::to_string(MaxWeight) + "])");
60 }
61 }
62 }
