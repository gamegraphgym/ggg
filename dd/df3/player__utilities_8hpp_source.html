96template <HasPlayerOnVertices GraphType>
97inline std::pair<std::set<typename boost::graph_traits<GraphType>::vertex_descriptor>,
98 std::map<typename boost::graph_traits<GraphType>::vertex_descriptor,
99 typename boost::graph_traits<GraphType>::vertex_descriptor>>
100compute_attractor(const GraphType &graph,
101 const std::set<typename boost::graph_traits<GraphType>::vertex_descriptor> &target,
102 int player) {
103 using VertexDescriptor = typename boost::graph_traits<GraphType>::vertex_descriptor;
104
105 std::set<VertexDescriptor> attractor = target;
106 std::map<VertexDescriptor, VertexDescriptor> strategy;
107 std::queue<VertexDescriptor> worklist;
108
109 for (const auto &vertex : target) {
110 worklist.push(vertex);
111 }
112
113 while (!worklist.empty()) {
114 const auto current = worklist.front();
115 worklist.pop();
116
117 // Walk all predecessors directly — O(in_degree(current)).
118 const auto [in_edges_begin, in_edges_end] = boost::in_edges(current, graph);
119 for (auto edge_it = in_edges_begin; edge_it != in_edges_end; ++edge_it) {
120 const auto predecessor = boost::source(*edge_it, graph);
121
122 if (attractor.count(predecessor)) {
123 continue;
124 }
125
126 bool should_add = false;
127
128 if (graph[predecessor].player == player) {
129 // Player vertex: this in-edge is sufficient to attract.
130 should_add = true;
131 strategy[predecessor] = current;
132 } else {
133 // Opponent vertex: attracted only when every successor is
134 // already in the attractor (no escape route).
135 const auto [out_edges_begin, out_edges_end] = boost::out_edges(predecessor, graph);
136 const auto all_to_attractor = std::all_of(out_edges_begin, out_edges_end, [&](const auto &edge) {
137 return attractor.count(boost::target(edge, graph));
138 });
139
140 if (all_to_attractor && boost::out_degree(predecessor, graph) > 0) {
141 should_add = true;
142 strategy[predecessor] = boost::target(*out_edges_begin, graph);
143 }
144 }
145
146 if (should_add) {
147 attractor.insert(predecessor);
148 worklist.push(predecessor);
149 }
150 }
151 }
152
153 return {attractor, strategy};
154}

167 std::vector<VertexDescriptor> result;
168

185 std::map<int, int> distribution;
186 const auto [vertices_begin, vertices_end] = boost::vertices(graph);
187 std::for_each(vertices_begin, vertices_end, [&graph, &distribution](const auto &vertex) {
188 distribution[graph[vertex].player]++;
