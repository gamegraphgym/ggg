63 std::uniform_int_distribution<int> weight_dist(min_weight, max_weight);
64 std::uniform_real_distribution<double> prob_dist(0.0, 1.0);
65 std::uniform_int_distribution<int> out_degree_dist(1, std::max(1, vertices - 1));
66
67 ggg::stochastic_discounted::graph::Graph graph;
68 std::vector<ggg::stochastic_discounted::graph::Vertex> vdesc;
69 vdesc.reserve(vertices);
70
71 for (int i = 0; i < vertices; ++i) {
72 const auto player = player_dist(gen);
73 const auto name = "v" + std::to_string(i);
74 vdesc.push_back(ggg::stochastic_discounted::graph::add_vertex(graph, name, player));
75 }
76
77 // For each vertex add a probabilistic distribution over targets with total probability <= 1
78 for (int i = 0; i < vertices; ++i) {
79 const auto out_degree = out_degree_dist(gen);
80 std::vector<int> targets(vertices);
81 std::iota(targets.begin(), targets.end(), 0);
82 std::shuffle(targets.begin(), targets.end(), gen);
83
84 double remaining = 1.0;
85 for (int k = 0; k < out_degree && remaining > 0.0; ++k) {
86 const auto target = targets[k];
87 const auto weight = weight_dist(gen);
88
89 // assign a probability for this transition; use the provided base probability as a hint
90 const double p = std::min(remaining, probability * prob_dist(gen));
91 remaining -= p;
92
