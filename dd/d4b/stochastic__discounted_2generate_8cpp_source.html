150 static ggg::stochastic_discounted::graph::Graph generate_stochastic_discounted_game(int player_vertices,
151 int min_player_outgoing,
152 int max_player_outgoing,
153 int min_weight,
154 int max_weight,
155 double discount,
156 std::mt19937 &gen) {
157 std::uniform_int_distribution<int> player_outgoing_dist(min_player_outgoing, max_player_outgoing);
158 std::vector<int> player_outdegrees(player_vertices);
159 int stochastic_vertices = 0;
160 for (auto &outdeg : player_outdegrees) {
161 outdeg = player_outgoing_dist(gen);
162 stochastic_vertices += outdeg;
163 }
164
165 const auto total_vertices = player_vertices + stochastic_vertices;
166 std::uniform_int_distribution<int> player_dist(0, 1);
167 std::uniform_int_distribution<int> weight_dist(min_weight, max_weight);
168 std::uniform_real_distribution<double> prob_dist(0.0, 1.0);
169
170 ggg::stochastic_discounted::graph::Graph graph;
171 std::vector<ggg::stochastic_discounted::graph::Vertex> vdesc;
172 vdesc.reserve(total_vertices);
173
174 // Create player-owned vertices (player 0 or 1, chosen at random)
175 for (int i = 0; i < player_vertices; ++i) {
176 vdesc.push_back(ggg::stochastic_discounted::graph::add_vertex(
177 graph, "v" + std::to_string(i), player_dist(gen)));
178 }
179
180 // Create stochastic vertices (player -1)
181 for (int i = 0; i < stochastic_vertices; ++i) {
182 vdesc.push_back(ggg::stochastic_discounted::graph::add_vertex(
183 graph, "v" + std::to_string(player_vertices + i), -1));
184 }
185
186 // Directed bipartite incidence matrices.
187 // Invariant for every pair (p, s): not both player_to_stoch[p][s] and stoch_to_player[s][p].
188 std::vector<std::vector<bool>> player_to_stoch(player_vertices,
189 std::vector<bool>(stochastic_vertices, false));
190 std::vector<std::vector<bool>> stoch_to_player(stochastic_vertices,
191 std::vector<bool>(player_vertices, false));
192
193 // Assign each stochastic vertex to exactly one owning player so that player outdegree
194 // requirements are met exactly and each stochastic vertex has at least N-1 admissible
195 // targets for outgoing stochastic->player edges.
196 std::vector<int> stochastic_owner;
197 stochastic_owner.reserve(stochastic_vertices);
198 for (int p = 0; p < player_vertices; ++p) {
199 for (int k = 0; k < player_outdegrees[p]; ++k) {
200 stochastic_owner.push_back(p);
201 }
202 }
203 std::shuffle(stochastic_owner.begin(), stochastic_owner.end(), gen);
204 for (int s = 0; s < stochastic_vertices; ++s) {
205 player_to_stoch[stochastic_owner[s]][s] = true;
206 }
207
208 // For each stochastic vertex, pick random outdegree in [1, N-1] and connect to player
209 // vertices that do not violate the no-reverse-edge invariant.
210 for (int s = 0; s < stochastic_vertices; ++s) {
211 std::vector<int> allowed;
212 allowed.reserve(player_vertices);
213 for (int p = 0; p < player_vertices; ++p) {
214 if (!player_to_stoch[p][s]) {
215 allowed.push_back(p);
216 }
217 }
218
219 if (allowed.size() != static_cast<std::size_t>(player_vertices - 1)) {
220 throw std::runtime_error("Internal generator error: unexpected admissible stochastic target count");
221 }
222
223 std::uniform_int_distribution<int> stoch_outdegree_dist(1, player_vertices - 1);
224 const int outdegree = stoch_outdegree_dist(gen);
225 std::shuffle(allowed.begin(), allowed.end(), gen);
226 for (int i = 0; i < outdegree; ++i) {
227 stoch_to_player[s][allowed[i]] = true;
228 }
229 }
230
231 // Materialize player->stochastic edges: weighted and discounted, non-probabilistic.
232 for (int p = 0; p < player_vertices; ++p) {
233 for (int s = 0; s < stochastic_vertices; ++s) {
234 if (player_to_stoch[p][s]) {
235 ggg::stochastic_discounted::graph::add_edge(
236 graph, vdesc[p], vdesc[player_vertices + s],
237 std::string(""), weight_dist(gen), discount, 0.0);
238 }
239 }
240 }
241
242 // Materialize stochastic->player edges: probabilities sum to exactly 1 per stochastic vertex.
243 for (int s = 0; s < stochastic_vertices; ++s) {
244 std::vector<int> targets;
245 targets.reserve(player_vertices);
246 for (int p = 0; p < player_vertices; ++p) {
247 if (stoch_to_player[s][p]) {
248 targets.push_back(p);
249 }
250 }
251
252 std::vector<double> probs(targets.size());
253 double total = 0.0;
254 for (auto &pr : probs) {
255 pr = prob_dist(gen);
256 total += pr;
257 }
258 double assigned = 0.0;
259 for (std::size_t i = 0; i + 1 < probs.size(); ++i) {
260 probs[i] /= total;
261 assigned += probs[i];
262 }
263 probs.back() = 1.0 - assigned;
264
265 for (std::size_t i = 0; i < targets.size(); ++i) {
266 ggg::stochastic_discounted::graph::add_edge(
267 graph, vdesc[player_vertices + s], vdesc[targets[i]],
268 std::string(""), 0, 0.0, probs[i]);
269 }
270 }
271
272 return graph;
273 }
