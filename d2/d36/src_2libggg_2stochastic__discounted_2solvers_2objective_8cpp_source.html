14bool StochasticDiscountedObjectiveSolver::switch_str(const graphs_t &graph) {
15 bool no_switch = true;
16 for (const auto &vertex : g::get_non_probabilistic_vertices(graph)) {

18 double oldval = graph[old_edge.first].weight;
19 const auto reach = g::get_reachable_through_probabilistic(
20 graph, vertex, strategy[vertex]);
21 for (const auto &[target, prob] : reach) {
22 oldval += prob * graph[old_edge.first].discount * sol[target];
23 }
24
25 const auto [out_edges_begin, out_edges_end] = boost::out_edges(vertex,
26 graph);
27 for (const auto &gedge :
28 boost::make_iterator_range(out_edges_begin, out_edges_end)) {
29 const auto successor = boost::target(gedge, graph);
30 if (successor == strategy[vertex]) {
31 continue;
32 }
33
34 const auto new_edge = boost::edge(vertex, successor, graph);
35 double newval = graph[new_edge.first].weight;
36 const auto reach2 = g::get_reachable_through_probabilistic(
37 graph, vertex, successor);
38 for (const auto &[target, prob] : reach2) {
39 newval += prob * graph[new_edge.first].discount * sol[target];
40 }
41
42 if (graph[vertex].player == 0) {
43 if (oldval + 1e-6 < newval) {
44 strategy[vertex] = successor;
45 switches++;
46 no_switch = false;
47 }
48 } else {
49 if (oldval > newval + 1e-6) {
50 strategy[vertex] = successor;
51 switches++;
52 no_switch = false;
53 }
54 }
55 }
56 }
