11bool StochasticDiscountedObjectiveSolver::switch_str(const graphs_t &graph) {
12 bool no_switch = true;
13 for (const auto &vertex : g::get_non_probabilistic_vertices(graph)) {
14 const auto olde = edge(vertex, strategy[vertex], graph);
15 double oldval = graph[olde.first].weight;
16 const auto reach = g::get_reachable_through_probabilistic(graph, vertex, strategy[vertex]);
17 for (const auto &[TARGET, PROB] : reach) {
18 oldval += PROB * graph[olde.first].discount * sol[TARGET];
19 }
20 const auto [out_edges_begin, out_edges_end] = boost::out_edges(vertex, graph);
21 for (const auto &gedge : boost::make_iterator_range(out_edges_begin, out_edges_end)) {
22 const auto successor = boost::target(gedge, graph);
23 if (successor != strategy[vertex]) {
24 const auto newe = edge(vertex, successor, graph);
25 double newval = graph[newe.first].weight;
26 const auto reach = g::get_reachable_through_probabilistic(graph, vertex, successor);
27 for (const auto &[TARGET, PROB] : reach) {
28 newval += PROB * graph[newe.first].discount * sol[TARGET];
29 }
30 if (((graph[vertex].player == 0) && (oldval + 1e-6 < newval)) || ((graph[vertex].player == 1) && (oldval > 1e-6 + newval))) { // Precision stop
31 strategy[vertex] = successor;
32 switches++;
33 no_switch = false;
34 }
35 }
36 }
37 }
38 return no_switch;

84 for (const auto &vertex : g::get_non_probabilistic_vertices(graph)) {
85 const auto curre = edge(vertex, strategy[vertex], graph);
86 const auto reach = g::get_reachable_through_probabilistic(graph, vertex, strategy[vertex]);
