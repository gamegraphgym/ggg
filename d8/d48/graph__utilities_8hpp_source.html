59struct OutDegreeValidator {

62 template <typename GraphType>
63 static void validate(const GraphType &graph) {
64 const auto [vertices_begin, vertices_end] = boost::vertices(graph);
65 for (const auto &vertex : boost::make_iterator_range(vertices_begin, vertices_end)) {
66 const auto out_deg = boost::out_degree(vertex, graph);
67 if constexpr (MinOutDegree > 0) {

93struct NoDuplicateEdgesValidator {
94 template <typename GraphType>
95 static void validate(const GraphType &graph) {
96 using Vertex = typename boost::graph_traits<GraphType>::vertex_descriptor;
97 std::set<std::pair<Vertex, Vertex>> seen_edges;
98 const auto [edges_begin, edges_end] = boost::edges(graph);
99 for (const auto &edge : boost::make_iterator_range(edges_begin, edges_end)) {
100 auto source = boost::source(edge, graph);
101 auto target = boost::target(edge, graph);
102 const auto edge_key = std::make_pair(source, target);
103 if (!seen_edges.insert(edge_key).second) {
104 throw GraphValidationError(
105 "Duplicate edge found between vertices '" +
106 graph[source].name + "' and '" + graph[target].name + "'");
107 }
108 }
109 }
