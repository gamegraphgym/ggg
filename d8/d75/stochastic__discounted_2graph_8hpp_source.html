56struct CycleValidator {
57 template <typename GraphType>
58 static void validate(const GraphType &graph) {
59 // Local vertex filter functor for player 1 vertices (default-constructible as required by Boost)
60 struct LocalPlayer1Filter {
61 const GraphType *graph_ptr;
62 LocalPlayer1Filter() : graph_ptr(nullptr) {}
63 explicit LocalPlayer1Filter(const GraphType &g) : graph_ptr(&g) {}
64 bool operator()(typename boost::graph_traits<GraphType>::vertex_descriptor v) const {
65 return graph_ptr && (*graph_ptr)[v].player == 1;
66 }
67 };

69 // Create filtered graph showing only player 1 vertices
70 auto filtered_graph = boost::make_filtered_graph(
71 graph,
72 boost::keep_all{},
73 LocalPlayer1Filter(graph));

75 // Local DFS visitor to detect back edges (cycles) without member templates
76 using FilteredGraphT = decltype(filtered_graph);
77 using FilteredEdgeT = typename boost::graph_traits<FilteredGraphT>::edge_descriptor;
78 struct LocalCycleDetector : public boost::dfs_visitor<> {
79 bool has_cycle = false;
80 void back_edge(FilteredEdgeT, const FilteredGraphT &) { has_cycle = true; }
81 } visitor;

85 throw graphs::GraphValidationError(

131inline double get_min_discount(const Graph &g) {
132 if (boost::num_edges(g) == 0) {
133 return 1.0;
134 }
135 const auto [eb, ee] = boost::edges(g);
136 const auto it = std::min_element(eb, ee, [&g](const auto &a, const auto &b) { return g[a].discount < g[b].discount; });
137 return g[*it].discount;

161inline std::map<Vertex, double> get_reachable_through_probabilistic(const Graph &g, Vertex source, Vertex successor) {
162 std::map<Vertex, double> reachable;
163 if (g[source].player == -1) {
164 return reachable;
165 }
166
167 std::queue<std::pair<Vertex, double>> q;
168 std::set<Vertex> visited;
169
170 if (g[successor].player == -1) {
171 q.push({successor, 1.0});
172 } else {
173 reachable[successor] = 1.0;
174 }
175
176 while (!q.empty()) {
177 auto [cur, prob] = q.front();
178 q.pop();
179 if (visited.count(cur)) {
180 continue;
181 }
182 visited.insert(cur);
183
184 const auto [ob, oe] = boost::out_edges(cur, g);
185 for (auto it = ob; it != oe; ++it) {
186 auto succ = boost::target(*it, g);
187 double edge_prob = g[*it].probability;
188 double total = prob * edge_prob;
189 if (g[succ].player == -1) {
190 if (!visited.count(succ)) {
191 q.push({succ, total});
192 }
193 } else {
194 reachable[succ] += total;
195 }
196 }
197 }
198 return reachable;
199}
