44struct DiscountValidator {
45 template <HasDiscountOnEdges GraphType, typename VertexFilter>
46 static void validate(const GraphType &graph, VertexFilter vertex_filter,
47 double min_discount = 0.0, double max_discount = 1.0) {
48 using Vertex = typename boost::graph_traits<GraphType>::vertex_descriptor;
49 const auto [edges_begin, edges_end] = boost::edges(graph);
50
51 for (const auto &edge : boost::make_iterator_range(edges_begin, edges_end)) {
52 auto source = boost::source(edge, graph);
53
54 // Only check discount for vertices matching the filter
55 if (vertex_filter(graph, source)) {
56 const double discount = graph[edge].discount;
57 if (discount <= min_discount || discount >= max_discount) {
58 throw GraphValidationError(
59 "Invalid discount factor " + std::to_string(discount) +
60 " on edge from vertex '" + graph[source].name +
61 "' (must be in range (" + std::to_string(min_discount) +
62 ", " + std::to_string(max_discount) + "))");
63 }
64 }
65 }
66 }
