179void PriorityPromotionSolver::attract(int priority) {
180 const int player = priority & 1;
181 auto &region_vertices = regions_[priority];
182
183 // If queue is empty, add all nodes in region to queue
184 if (queue_.empty()) {
185 for (Vertex v : region_vertices) {
186 queue_.push(v);
187 }
188 }
189
190 // Attract predecessors
191 while (!queue_.empty()) {
192 Vertex current = queue_.front();
193 queue_.pop();
194
195 // Check all predecessors of current vertex
196 for (auto in_edge : boost::make_iterator_range(boost::in_edges(current, *graph_))) {
197 Vertex pred = boost::source(in_edge, *graph_);
198 if (is_disabled(pred) || get_region(pred) > priority) {
199 continue; // Skip disabled or higher priority vertices
200 } else if (get_region(pred) == priority) {
201 // Already in region - update strategy if needed
202 if (get_player(pred) == player && !has_strategy(pred)) {
203 strategy_[pred] = current;
204 has_strategy_[pred] = true;
205 }
206 } else if (get_player(pred) == player) {
207 // Same player - attract to region
208 region_vertices.push_back(pred);
209 region_[pred] = priority;
210 strategy_[pred] = current;
211 has_strategy_[pred] = true;
212 queue_.push(pred);
213 } else {
214 // Opponent player - check if forced (all successors in region or higher)
215 bool can_escape = false;
216 for (auto out_edge : boost::make_iterator_range(boost::out_edges(pred, *graph_))) {
217 Vertex succ = boost::target(out_edge, *graph_);
218 if (!is_disabled(succ) && get_region(succ) < priority) {
219 can_escape = true;
220 break;
221 }
222 }
223
224 if (!can_escape) {
225 // All successors are in region or higher - attract
226 region_vertices.push_back(pred);
227 region_[pred] = priority;
228 has_strategy_[pred] = false; // No strategy for opponent
229 queue_.push(pred);
230 }
231 }
232 }
233 }

236void PriorityPromotionSolver::promote(int from_priority, int to_priority) {
237 assert(from_priority < to_priority);

272bool PriorityPromotionSolver::setup_region(int vertex_index, int priority, bool must_reset) {
273 // PP always resets (must_reset is always true in PP)
274 if (!regions_[priority].empty()) {
275 reset_region(priority);
276 }
277
278 // Add all escapes (vertices with original priority p that should be in the region)
279 for (int j = vertex_index; j < static_cast<int>(sorted_vertices_.size()) &&
280 get_original_priority(sorted_vertices_[j]) == priority;
281 j++) {
282 Vertex v = sorted_vertices_[j];
283
284 if (is_disabled(v)) {
285 region_[v] = -2; // set to disabled
286 } else if (get_region(v) == get_original_priority(v)) {
287 // Vertex is at its original priority level, should be in region
288 regions_[priority].push_back(v);
289 region_[v] = priority;
290 has_strategy_[v] = false; // Reset strategy
291 }
292 }
293
294 // If region is empty, return false
295 if (regions_[priority].empty()) {
296 return false;
297 }
298
299 // Attract to maximize the region
300 attract(priority);
301 return true;

304void PriorityPromotionSolver::set_dominion(int priority, Solution &solution) {
305 const int player = priority & 1;
306 auto &region_vertices = regions_[priority];
307
308 // Create a filtered graph view that only contains non-disabled vertices
309 // For simplicity, we'll use a different approach: manual attractor computation
310 // on the implicitly filtered graph
311
312 std::set<Vertex> dominion_core;
313 std::set<Vertex> full_attractor;
314 std::map<Vertex, Vertex> strategy_map;
315 doms++; // Increment dominions counter
316
317 // Step 1: Collect initial vertices in the dominion (current region)
318 for (Vertex v : region_vertices) {
319 if (get_region(v) == priority && !is_disabled(v)) {
320 dominion_core.insert(v);
321 full_attractor.insert(v);
322 }
323 }
324
325 // Step 2: Manually compute attractor, respecting disabled vertices
326 std::queue<Vertex> worklist;
327 for (Vertex v : dominion_core) {
328 worklist.push(v);
329 }
330
331 while (!worklist.empty()) {
332 Vertex current = worklist.front();
333 worklist.pop();
334
335 // Check all non-disabled predecessors
336 for (auto in_edge : boost::make_iterator_range(boost::in_edges(current, *graph_))) {
337 Vertex pred = boost::source(in_edge, *graph_);
338 if (is_disabled(pred) || full_attractor.count(pred)) {
339 continue; // Skip disabled or already in attractor
340 }
341
342 bool should_add = false;
343
344 if (get_player(pred) == player) {
345 // Player vertex: add if has edge to attractor
346 should_add = true;
347 strategy_map[pred] = current;
348 } else {
349 // Opponent vertex: add if all non-disabled edges lead to attractor
350 bool all_to_attractor = true;
351 bool has_edges = false;
352
353 for (auto out_edge : boost::make_iterator_range(boost::out_edges(pred, *graph_))) {
354 Vertex succ = boost::target(out_edge, *graph_);
355 if (is_disabled(succ))
356 continue;
357 has_edges = true;
358 if (!full_attractor.count(succ)) {
359 all_to_attractor = false;
360 break;
361 }
362 }
363
364 if (has_edges && all_to_attractor) {
365 should_add = true;
366 // For opponent, choose any edge to attractor
367 for (auto out_edge : boost::make_iterator_range(boost::out_edges(pred, *graph_))) {
368 Vertex succ = boost::target(out_edge, *graph_);
369 if (!is_disabled(succ) && full_attractor.count(succ)) {
370 strategy_map[pred] = succ;
371 break;
372 }
373 }
374 }
375 }
376
377 if (should_add) {
378 full_attractor.insert(pred);
379 worklist.push(pred);
380 }
381 }
382 }
383
384 // Step 3: Mark all vertices in the full attractor as won by the player
385 for (Vertex v : full_attractor) {
386 solution.set_winning_player(v, player);
387 disabled_[v] = true;
388 }
389
390 // Step 4: Set strategies from attractor computation
391 // Only set strategies for vertices owned by the winning player
392 for (const auto &[from, to] : strategy_map) {
393 int vertex_owner = get_player(from);
394 // PLAYER is the winning player for this dominion
395 if (vertex_owner == player) {
396 solution.set_strategy(from, to);
397 }
398 // If vertex_owner != PLAYER, then this vertex is owned by the opponent
399 // and should not have a strategy set
400 }

403int PriorityPromotionSolver::get_region_status(int vertex_index, int priority) {
404 const int player = priority & 1;
405
406 // Check if the region is closed in the subgame
407 for (int j = vertex_index; j < static_cast<int>(sorted_vertices_.size()) &&
408 get_original_priority(sorted_vertices_[j]) == priority;
409 j++) {
410 Vertex v = sorted_vertices_[j];
411
412 if (is_disabled(v) || get_region(v) > priority) {
413 continue; // Skip disabled or higher regions
414 }
415
416 if (get_player(v) == player) {
417 // Region player owns this vertex - must have a strategy
418 if (!has_strategy(v)) {
419 return -2; // Region is open
420 }
421 } else {
422 // Opponent owns this vertex - check for escapes to lower regions
423 for (auto out_edge : boost::make_iterator_range(boost::out_edges(v, *graph_))) {
424 Vertex succ = boost::target(out_edge, *graph_);
425 if (is_disabled(succ))
426 continue;
427 if (get_region(succ) < priority) {
428 return -2; // Open - escape to lower region
429 }
430 }
431 }
432 }
433
434 // Region is closed, find promotion target (lowest higher region)
435 int lowest_promotion_target = -1;
436
437 // Check for promotion opportunities (exits to higher regions)
438 for (Vertex v : regions_[priority]) {
439 if (get_player(v) != player) {
440 // Check successors for higher regions
441 for (auto out_edge : boost::make_iterator_range(boost::out_edges(v, *graph_))) {
442 Vertex succ = boost::target(out_edge, *graph_);
443 if (is_disabled(succ))
444 continue;
445 int succ_region = get_region(succ);
446 if (succ_region > priority &&
447 (succ_region < lowest_promotion_target || lowest_promotion_target == -1)) {
448 lowest_promotion_target = succ_region;
449 }
450 }
451 }
452 }
453
454 return lowest_promotion_target; // -1 if dominion, target region otherwise
455}
