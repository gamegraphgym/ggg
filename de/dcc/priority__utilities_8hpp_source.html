37template <int MinPriority = 0, int MaxPriority = INT_MAX>
38struct PriorityValidator {
39 template <HasPriorityOnVertices GraphType>
40 static void validate(const GraphType &graph) {
41 const auto [vertices_begin, vertices_end] = boost::vertices(graph);
42 for (const auto &vertex : boost::make_iterator_range(vertices_begin, vertices_end)) {
43 const int priority = graph[vertex].priority;
44 if (priority < MinPriority || priority > MaxPriority) {
45 throw GraphValidationError(
46 "Invalid priority " + std::to_string(priority) +
47 " for vertex '" + graph[vertex].name +
48 "' (must be in range [" + std::to_string(MinPriority) +
49 ", " + std::to_string(MaxPriority) + "])");
50 }
51 }
52 }
53};

173template <HasPriorityOnVertices GraphType>
174inline void compress_priorities(GraphType &graph) {
175 // Get all unique priorities and sort them
176 auto unique_priorities = get_unique_priorities(graph);
177
178 if (unique_priorities.empty()) {
179 return; // Nothing to compress
180 }
181
182 // Determine starting values based on the parity of the minimum priority
183 int min_priority = unique_priorities[0];
184 int min_parity = min_priority % 2;
185
186 // Start compressed priorities to preserve the parity of the minimum
187 int next_even, next_odd;
188 if (min_parity == 0) { // Minimum is even
189 next_even = 0; // Start even sequence at 0
190 next_odd = 1; // Start odd sequence at 1
191 } else { // Minimum is odd
192 next_even = 2; // Start even sequence at 2
193 next_odd = 1; // Start odd sequence at 1
194 }
195
196 // Create the compression mapping with parity preservation
197 std::map<int, int> mapping;
198 int current_even = -1; // Current even priority being used
199 int current_odd = -1; // Current odd priority being used
200 int last_parity = -1; // Parity of the last processed priority
201
202 for (int priority : unique_priorities) {
203 int parity = priority % 2; // 0 for even, 1 for odd
204
205 if (parity == 0) { // Even priority
206 // Use new even priority if this is the first even or if we had an odd priority since last even
207 if (current_even == -1 || last_parity == 1) {
208 current_even = next_even;
209 next_even += 2; // Next even number
210 }
211 mapping[priority] = current_even;
212 } else { // Odd priority
213 // Use new odd priority if this is the first odd or if we had an even priority since last odd
214 if (current_odd == -1 || last_parity == 0) {
215 current_odd = next_odd;
216 next_odd += 2; // Next odd number
217 }
218 mapping[priority] = current_odd;
219 }
220
221 last_parity = parity;
222 }
223
224 // Update all vertex priorities in-place
225 const auto [VERTICES_BEGIN, VERTICES_END] = boost::vertices(graph);
226 for (auto vertex_it = VERTICES_BEGIN; vertex_it != VERTICES_END; ++vertex_it) {
227 const auto &vertex = *vertex_it;
228 int old_priority = graph[vertex].priority;
229 graph[vertex].priority = mapping[old_priority];
230 }
231}
